from keras.preprocessing.image import ImageDataGenerator

### Data augmentation for training set ###
# rotation by 40-degree
# shift horizontally and vertically
# shear
# zoom
# flip
train_datagen = ImageDataGenerator(rescale=1./255,    
                                   rotation_range=40,
                                   width_shift_range=0.2,
                                   height_shift_range=0.2, 
                                   shear_range=0.2,
                                   zoom_range=0.2,
                                   horizontal_flip=True,)

validation_datagen = ImageDataGenerator(rescale=1./255)
test_datagen = ImageDataGenerator(rescale=1./255)

# Create a Generator object for training dataset
train_generator = train_datagen.flow_from_directory(
                train_dir,
                target_size=(150, 150),
                batch_size=20,
                class_mode='binary')

# Create a Generator object for validation dataset
validation_generator = validation_datagen.flow_from_directory(
                validation_dir,
                target_size=(150, 150),
                batch_size=20,
                class_mode='binary')

# Create a Generator object for test dataset
test_generator = test_datagen.flow_from_directory(
                test_dir,
                target_size=(150, 150),
                batch_size=20,
                class_mode='binary')

Build model¶

convnet

  1. Convolutional Layer 1. ReLU filters 3x3 32 , 150x150x3, 148x148x32
  2. Max Pooling Layer 1. ReLU filters 2x2 32 , 74x74x32
  3. Convolutional Layer 2. filters 3x3x32 64 , 72x72x64
  4. Max Pooling Layer 2. filters 2x2 64 , 36x36x64
  5. Convolutional Layer 3. filters 3x3x64 128 , 34x34x128
  6. Max Pooling Layer 3. filters 2x2 128 , 17x17x128
  7. Convolutional Layer 4. filters 3x3x128 128 , 15x15x128
  8. Max Pooling Layer 4. filters 2x2 128 , 7x7x128
  9. Flatten Layer. 7x7x128 6272x1
  10. Dropout Layer. Dropout overfitting Dense Layer ( 0.5)
  11. Dense Layer 1. ReLU 512 fully connected
  12. Output Layer. sigmoid 1
In [8]:
from keras import layers
from keras import models

model = models.Sequential()
model.add(layers.Conv2D(32, (3, 3), activation='relu', 
                        input_shape=(150, 150, 3)))
model.add(layers.MaxPool2D((2, 2)))
model.add(layers.Conv2D(64, (3, 3), activation='relu'))
model.add(layers.MaxPool2D((2, 2)))
model.add(layers.Conv2D(128,(3, 3), activation='relu'))
model.add(layers.MaxPool2D((2, 2)))
model.add(layers.Conv2D(128,(3, 3), activation='relu'))
model.add(layers.MaxPool2D((2, 2)))
model.add(layers.Flatten())
model.add(layers.Dropout(0.5))
model.add(layers.Dense(512, activation='relu'))
model.add(layers.Dense(1, activation='sigmoid'))
